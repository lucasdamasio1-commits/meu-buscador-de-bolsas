name: Atualizacao Diaria de Bolsas
on:
  schedule:
    - cron: '0 9 * * *' # 09:00 UTC = 06:00 Brasília [1]
  workflow_dispatch: # Permite que você clique num botão e rode na hora para testar

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Baixar codigo do repositorio
        uses: actions/checkout@v4 [2]

      - name: Instalar Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.10' [1]

      - name: Instalar Dependencias
        run: |
          pip install -r requirements.txt
          playwright install chromium

      - name: Executar Script Principal
        env:
          # O GitHub vai buscar esses valores nos "Secrets" que vamos configurar
          SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
          SUPABASE_KEY: ${{ secrets.SUPABASE_KEY }}
        run: python main.py
